\newcommand{\carloCubeBullets}{

\foreach \x in {0,1}
 \foreach \y in {0,1}
  \foreach \z in {0,1}
   \node (\x\y\z) at (\x , \y , \z) {\textbullet} ;
\end{scope}\end{pgfonlayer}}

\end{scope}\end{pgfonlayer}}
